به منظور نصب فونت ابتدا یک پوشه را به وسیله دستور زیر بسازید


mkdir /usr/share/fonts/myfonts


chmod 775 -R /usr/share/fonts/myfonts


حال فونت های خود را به آن منتقل کرده و دستورات زیر را اجرا نمایید
cd /usr/share/fonts
fc-cache myfonts
